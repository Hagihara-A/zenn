from typing import List
T = TypeVar("T")


def ES(seq: List[T]):
    """exhaustive search"""
    if(len(seq) == 1):
        yield []
        yield seq
    else:
        head, rest = seq[0], seq[1:]
        for cand in ES(rest):
            yield cand
            yield [head] + cand


for c in ES([1, 2, 3, 4]):
    print(c)
